package 满分标准; import java.util.ArrayList; import java.util.Scanner;

public class Ja1011 { public static void main(String[] args) { Scanner scan = new Scanner(System.in); ArrayList list = new ArrayList(); long a,b,c; int i=1; int count = scan.nextInt(); while(count > 0){ a = scan.nextLong(); b = scan.nextLong(); c = scan.nextLong(); if(a+b>c)list.add("true"); else list.add("false"); count --; } scan.close(); for(String str:list){ System.out.println("Case #"+i+": "+str); i++; }

}

}
